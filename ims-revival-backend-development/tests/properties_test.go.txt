func TestCreateEmpPropertyDetails_Success(t *testing.T) {
	input := `{
		"properties": [
			{
				"employee_id": 10015845,
				"employee_name": "Anand Kumar",
				"group_name": "Group A",
				"property_type": "Residential",
				"property_details": "3 BHK flat",
				"location_details": "Hyderabad, India",
				"house_construction_or_purchase_cost": 5000000,
				"house_construction_or_purchase_year": 2020,
				"present_value": 6000000,
				"holder_name": "Anand Kumar",
				"holder_relationship": "Self",
				"acquisition_method": "Purchased",
				"acquisition_year": 2020,
				"annual_income_from_property": 50000,
				"statement_year": 2024,
				"statement_as_on": "2024-01-01T00:00:00Z",
				"approver_post_id": "30007706",
				"created_by": "10021629",
				"remarks": "Property details for test",
				"employee_designation": "Postal Assistant",
				"basic_pay": 50000,
				"office_of_working": "Hyderabad Office",
				"date_of_join_in_department": "2020-01-01T00:00:00Z"
			}
		]
	}`
	req := httptest.NewRequest("POST", "/v1/properties", bytes.NewBuffer([]byte(input)))
	req.Header.Set(contentType, applicationJson)
	rec := httptest.NewRecorder()
	Engine.ServeHTTP(rec, req)
	assert.Equal(t, http.StatusOK, rec.Code)
}
