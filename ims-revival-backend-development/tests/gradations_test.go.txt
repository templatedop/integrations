package tests

func TestUpdateGradations_UriBindingError(t *testing.T) {
	input := `{
	 "remarks": "Verified",
  	 "ack_status": "Acknowledged"
	}`
	req := httptest.NewRequest("PUT", "/v1/gradations/abc/acknowledge", bytes.NewBuffer([]byte(input)))
	req.Header.Set(contentType, applicationJson)
	rec := httptest.NewRecorder()
	Engine.ServeHTTP(rec, req)

	assert.Equal(t, http.StatusBadRequest, rec.Code)
}

func TestUpdateGradations_Success(t *testing.T) {
	// Define the JSON input for the test
	input := `{
		"ack_status": "Acknowledged",
		"remarks": "Gradation updated successfully"
	}`

	// Create a test request
	req := httptest.NewRequest("PUT", "/v1/gradations/10015845/acknowledge", bytes.NewBuffer([]byte(input)))
	req.Header.Set(contentType, applicationJson)
	// Create a test recorder to capture the response
	rec := httptest.NewRecorder()
	// Simulate the API call
	Engine.ServeHTTP(rec, req)
	// Assert that the response code is 200 OK
	assert.Equal(t, http.StatusOK, rec.Code)
}
