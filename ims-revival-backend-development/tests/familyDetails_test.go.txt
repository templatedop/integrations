func TestCreateFamilyDetails_Success(t *testing.T) {
	filePath := "../tests/files/example.pdf"
	file, err := os.Open(filePath)
	assert.NoError(t, err)
	defer file.Close()

	body := &bytes.Buffer{}
	writer := multipart.NewWriter(body)
	input := `{
    "family": [
        {
            "employee_id": 10035071,
            "dependent_name": "Dhruva",
            "relation": "Son",
            "date_of_birth": "2010-06-15T00:00:00Z",
            "marital_status": "UnMarried",
            "highest_education_qualification": "High School",
            "employment_status": "Student",
            "dependent_status": "Active",
            "approver_post_id": "30007775",
            "created_by": "10035071",
            "user_remarks": "No remarks",
            "gender": "Male",
            "pwd": false,
            "pwd_percentage": 0.0,
            "relation_id": 4,
            "file_name":"example.pdf"
        },
        {
            "employee_id": 10035071,
            "dependent_name": "Danvika",
            "relation": "Daughter",
            "date_of_birth": "2012-08-20T00:00:00Z",
            "marital_status": "UnMarried",
            "highest_education_qualification": "Primary School",
            "employment_status": "Student",
            "dependent_status": "Active",
            "approver_post_id": "30007775",
            "created_by": "10035071",
            "user_remarks": "No remarks",
            "gender": "Female",
            "pwd": true,
            "pwd_percentage": 20.5,
            "relation_id": 5,
            "file_name":"example.pdf"
        }
    ]
}`
	_ = writer.WriteField("data", input)

	part1, err := writer.CreateFormFile("files", filePath)
	assert.NoError(t, err)
	_, err = io.Copy(part1, file)
	assert.NoError(t, err)

	part2, err := writer.CreateFormFile("files", filePath)
	assert.NoError(t, err)
	_, err = io.Copy(part2, file)
	assert.NoError(t, err)
	_ = writer.Close()
	req := httptest.NewRequest("POST", "/v1/family-details-makers", body)
	req.Header.Set(contentType, writer.FormDataContentType())
	rec := httptest.NewRecorder()
	Engine.ServeHTTP(rec, req)
	assert.Equal(t, http.StatusOK, rec.Code)
}
