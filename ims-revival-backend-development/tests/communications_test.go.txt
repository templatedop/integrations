func TestCreateCommunications_Success(t *testing.T) {
	filePath := "../tests/files/example.pdf"
	file, err := os.Open(filePath)
	assert.NoError(t, err)
	defer file.Close()

	body := &bytes.Buffer{}
	writer := multipart.NewWriter(body)
	input := `{
    "employee_id": 10015845,
    "communication_addr1": "1234 Main St",
    "communication_addr2": "Apt 101",
    "communication_addr3": "Near Central Park",
    "communication_pin": 110001,
    "india_post_email_id": "employee@indiapost.gov.in",
    "personal_email_id": "employee.personal@gmail.com",
    "mobile_no": "9876543210",
    "aadhaar_ref_number": "1234-5678-9012",
    "pan_number": "ABCDE1234F",
    "approver_post_id": "30007775",
    "created_by": "10015845",
    "user_remarks": "Some remarks"
}`
	_ = writer.WriteField("data", input)

	part, err := writer.CreateFormFile("file", filePath)
	assert.NoError(t, err)
	_, err = io.Copy(part, file)
	assert.NoError(t, err)

	_ = writer.Close()

	req := httptest.NewRequest("POST", "/v1/communication-makers", body)
	req.Header.Set(contentType, writer.FormDataContentType())
	rec := httptest.NewRecorder()
	Engine.ServeHTTP(rec, req)
	assert.Equal(t, http.StatusOK, rec.Code)
}
