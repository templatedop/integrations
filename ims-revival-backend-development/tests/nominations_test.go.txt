func TestCreateBulkNominations_Success(t *testing.T) {
	filePath := "../tests/files/example.pdf"
	file, err := os.Open(filePath)
	assert.NoError(t, err)
	defer file.Close()

	body := &bytes.Buffer{}
	writer := multipart.NewWriter(body)
	input := `{
    "nominations": [
        {
            "employee_id": 10133660,
            "employee_name": "KISHORE",
            "type_of_nomination": "Pension",
            "nomination_type_id": 2,
            "nominee_name": "Anjali",
            "relation": "Mother",
            "date_of_birth": "1976-01-01T00:00:00Z",
            "nomination_addr1": "address1",
            "nomination_addr2": "adress2",
            "nomination_pin": 500001,
            "share_percentage": 50,
            "approver_post_id": "30007706",
            "forward_post_id": "30007706",
            "created_by": "10133660",
            "remarks": "",
            "file_name":"example.pdf"
        },
        {
            "employee_id": 10133660,
            "employee_name": "KISHORE",
            "type_of_nomination": "Pension",
            "nomination_type_id": 2,
            "nominee_name": "Anirudh",
            "relation": "Father",
            "date_of_birth": "1970-01-01T00:00:00Z",
            "nomination_addr1": "address1",
            "nomination_addr2": "addreess2",
            "nomination_pin": 500001,
            "share_percentage": 50,
            "approver_post_id": "30007706",
            "forward_post_id": "30007706",
            "created_by": "10133660",
            "remarks": "",
            "file_name":"example.pdf"
        }
    ]
}`
	_ = writer.WriteField("data", input)

	part1, err := writer.CreateFormFile("files", filePath)
	assert.NoError(t, err)
	_, err = io.Copy(part1, file)
	assert.NoError(t, err)

	part2, err := writer.CreateFormFile("files", filePath)
	assert.NoError(t, err)
	_, err = io.Copy(part2, file)
	assert.NoError(t, err)
	_ = writer.Close()
	req := httptest.NewRequest("POST", "/v1/nomination-makers", body)
	req.Header.Set(contentType, writer.FormDataContentType())
	rec := httptest.NewRecorder()
	Engine.ServeHTTP(rec, req)
	assert.Equal(t, http.StatusOK, rec.Code)
}
