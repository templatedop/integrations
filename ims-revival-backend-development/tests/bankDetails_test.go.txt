func TestCreateBankDetailsBulk_Success(t *testing.T) {
	filePath := "../tests/files/example.pdf"
	file, err := os.Open(filePath)
	assert.NoError(t, err)
	defer file.Close()

	body := &bytes.Buffer{}
	writer := multipart.NewWriter(body)
	input := `{
    "banks": [
        {
            "employee_id": 10035071,
            "bank_name": "SBI",
            "branch_code": "500007",
            "bank_acc_no": "547369345680",
            "ifsc_code": "SBIN002315",
            "branch_addr1": "123-1 Abids",
            "branch_addr2": "Hyderabad",
            "branch_pin": 500001,
            "approver_post_id": "30007706",
            "bank_type": "Home",
            "draw_type": "Main bank",
            "created_by": "10035071",
            "user_remarks": "sample remarks",
            "file_name": "example.pdf"
        },
        {
            "employee_id": 10035071,
            "bank_name": "POSB",
            "branch_code": "500007",
            "bank_acc_no": "3313456789",
            "ifsc_code": "IPOS0000DOP",
            "branch_addr1": "123-1 Abids",
            "branch_addr2": "Hyderabad",
            "branch_pin": 500001,
            "approver_post_id": "30007706",
            "bank_type": "POSB",
            "draw_type": "Main bank details for Off-Cycle",
            "created_by": "10035071",
            "user_remarks": "sample remarks",
            "file_name": "example.pdf"
        }
    ]
}`
	_ = writer.WriteField("data", input)

	part1, err := writer.CreateFormFile("files", filePath)
	assert.NoError(t, err)
	_, err = io.Copy(part1, file)
	assert.NoError(t, err)

	part2, err := writer.CreateFormFile("files", filePath)
	assert.NoError(t, err)
	_, err = io.Copy(part2, file)
	assert.NoError(t, err)
	_, err = io.Copy(part1, file)
	assert.NoError(t, err)

	_ = writer.Close()

	req := httptest.NewRequest("POST", "/v1/bank-detail-makers", body)
	req.Header.Set(contentType, writer.FormDataContentType())
	rec := httptest.NewRecorder()
	Engine.ServeHTTP(rec, req)
	assert.Equal(t, http.StatusOK, rec.Code)
}
