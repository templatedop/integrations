// CreateEducationsBulk test cases
func TestCreateEducationsBulk_Success(t *testing.T) {
	filePath := "../tests/files/example.pdf"
	file, err := os.Open(filePath)
	assert.NoError(t, err)
	defer file.Close()

	body := &bytes.Buffer{}
	writer := multipart.NewWriter(body)
	input := `{
    "qualifications": [
        {
            "employee_id": 10015845,
            "education_qualification": "Bachelor of Science",
            "board_studied": "University of XYZ",
            "year_of_study": 2015,
            "state_of_study": "Telangana",
            "marks_grade_percentage": 85.6,
            "approver_post_id": "30007775",
            "created_by": "10015845",
            "computer_knowledge": "F",
            "user_remarks": "Good academic record",
            "file_name":"example.pdf"
        },
        {
            "employee_id": 10035071,
            "education_qualification": "Master of Engineering",
            "board_studied": "Institute of ABC",
            "year_of_study": 2018,
            "state_of_study": "Karnataka",
            "marks_grade_percentage": 90.2,
            "approver_post_id": "30007775",
            "created_by": "10015845",
            "computer_knowledge": "T",
            "user_remarks": "Highly qualified in software development",
            "file_name":"example.pdf"
        }
    ]
}`
	_ = writer.WriteField("data", input)

	part1, err := writer.CreateFormFile("files", filePath)
	assert.NoError(t, err)
	_, err = io.Copy(part1, file)
	assert.NoError(t, err)

	part2, err := writer.CreateFormFile("files", filePath)
	assert.NoError(t, err)
	_, err = io.Copy(part2, file)
	assert.NoError(t, err)
	_ = writer.Close()
	req := httptest.NewRequest("POST", "/v1/education-makers", body)
	req.Header.Set(contentType, writer.FormDataContentType())
	rec := httptest.NewRecorder()
	Engine.ServeHTTP(rec, req)
	assert.Equal(t, http.StatusOK, rec.Code)
}
