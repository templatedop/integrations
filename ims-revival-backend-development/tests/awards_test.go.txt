func TestCreateAwardsBulk_Success(t *testing.T) {
	filePath := "../tests/files/example.pdf"
	file, err := os.Open(filePath)
	assert.NoError(t, err)
	defer file.Close()

	body := &bytes.Buffer{}
	writer := multipart.NewWriter(body)
	input := `{
    "awards": [
        {
            "employee_id": 10015845,
            "award_name": "Best Employee of the Year",
            "award_type": "Annual",
            "award_category": "Performance",
            "award_description": "Awarded for outstanding performance throughout the year",
            "certificate_no": "CERT-5678",
            "monetary_benefit": 1000.50,
            "approver_post_id": "30007775",
            "created_by": "10015845",
            "award_issue_date": "2023-05-15T00:00:00Z",
            "award_for_year": 2022,
            "user_remarks": "Exceptional leadership in project delivery",
            "award_id": 48,
            "file_name":"example.pdf"
        },
        {
            "employee_id": 10015845,
            "award_name": "Innovation Award",
            "award_type": "Quarterly",
            "award_category": "Innovation",
            "award_description": "Awarded for innovative project execution",
            "certificate_no": "CERT-9101",
            "monetary_benefit": 500.00,
            "approver_post_id": "30007775",
            "created_by": "10015845",
            "award_issue_date": "2024-06-20T00:00:00Z",
            "award_for_year": 2023,
            "user_remarks": "Developed a new workflow automation",
            "file_name":"example.pdf"
        }
    ]
}`
	_ = writer.WriteField("data", input)

	part1, err := writer.CreateFormFile("files", filePath)
	assert.NoError(t, err)
	_, err = io.Copy(part1, file)
	assert.NoError(t, err)

	part2, err := writer.CreateFormFile("files", filePath)
	assert.NoError(t, err)
	_, err = io.Copy(part2, file)
	assert.NoError(t, err)
	_, err = io.Copy(part1, file)
	assert.NoError(t, err)

	_ = writer.Close()

	req := httptest.NewRequest("POST", "/v1/award-makers", body)
	req.Header.Set(contentType, writer.FormDataContentType())
	rec := httptest.NewRecorder()
	Engine.ServeHTTP(rec, req)
	assert.Equal(t, http.StatusOK, rec.Code)
}
