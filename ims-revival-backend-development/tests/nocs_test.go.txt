func TestCreateEmpNocDetail_Success(t *testing.T) {
	filePath := "../tests/files/example.pdf"
	file, err := os.Open(filePath)
	assert.NoError(t, err)
	defer file.Close()
	var input string
	body := &bytes.Buffer{}
	writer := multipart.NewWriter(body)
	input = `{
  "employee_id": 10015845,
  "noc_id": 2,
  "noc_purpose": "Higher Studies",
  "country_name": "United States",
  "noc_from_date": "2024-07-01T00:00:00Z",
  "noc_to_date": "2024-12-31T00:00:00Z",
  "course_name": "Masters in Computer Science",
  "course_duration": "2 years",
  "institution_name": "Massachusetts Institute of Technology",
  "university_name": "MIT",
  "institution_location": "Cambridge, MA",
  "fwd_post_id": "30005933",
  "approver_post_id": "30007775",
  "created_by": "10015845",
  "user_remarks": "Application for NOC for higher studies in the US."
}`
	_ = writer.WriteField("data", input)

	part, err := writer.CreateFormFile("file", filePath)
	assert.NoError(t, err)
	_, err = io.Copy(part, file)
	assert.NoError(t, err)

	_ = writer.Close()

	req := httptest.NewRequest("POST", "/v1/nocs", body)
	req.Header.Set(contentType, writer.FormDataContentType())
	rec := httptest.NewRecorder()
	Engine.ServeHTTP(rec, req)
	assert.Equal(t, http.StatusOK, rec.Code)
}
