CREATE OR REPLACE FUNCTION collection.generate_receipt_number()
 RETURNS trigger
 LANGUAGE plpgsql
AS $function$
BEGIN
    IF NEW.receipt_number IS NULL THEN
        NEW.receipt_number := 'RCP' || LPAD(nextval('collection.receipt_number_seq')::TEXT, 10, '0');
    END IF;
    RETURN NEW;
END;
$function$
;

-- DROP FUNCTION collection.validate_suspense_reversal();

CREATE OR REPLACE FUNCTION collection.validate_suspense_reversal()
 RETURNS trigger
 LANGUAGE plpgsql
AS $function$
DECLARE
    v_is_first_collection BOOLEAN;
    v_collection_batch_id UUID;
BEGIN
    -- Check if this suspense is from first collection
    SELECT DISTINCT pt.collection_batch_id INTO v_collection_batch_id
    FROM collection.payment_transactions pt
    WHERE pt.payment_id = ANY(NEW.source_payment_ids)
    AND pt.collection_batch_id IS NOT NULL;
    
    v_is_first_collection := (v_collection_batch_id IS NOT NULL);
    
    -- IR_28: Block reversal of first collection suspense
    IF NEW.is_reversed = true AND v_is_first_collection THEN
        RAISE EXCEPTION 'IR_28 Violation: Suspense reversal not allowed for first collection'
            USING ERRCODE = '23514';
    END IF;
    
    RETURN NEW;
END;
$function$;

-- collection.collection_status_summary source

CREATE OR REPLACE VIEW collection.collection_status_summary
AS SELECT cbt.request_id,
    cbt.policy_number,
    cbt.collection_complete,
    cbt.collection_date,
    pt_premium.payment_id AS premium_payment_id,
    pt_premium.payment_status AS premium_status,
    pt_installment.payment_id AS installment_payment_id,
    pt_installment.payment_status AS installment_status,
    cbt.combined_receipt_id
   FROM collection.collection_batch_tracking cbt
     LEFT JOIN collection.payment_transactions pt_premium ON cbt.premium_payment_id = pt_premium.payment_id
     LEFT JOIN collection.payment_transactions pt_installment ON cbt.installment_payment_id = pt_installment.payment_id;

COMMENT ON VIEW collection.collection_status_summary IS 'First collection status tracking (IR_34)';


-- collection.pending_cheque_clearances source

CREATE OR REPLACE VIEW collection.pending_cheque_clearances
AS SELECT cheque_id,
    request_id,
    policy_number,
    cheque_number,
    bank_name,
    amount,
    created_at,
    next_due_date,
        CASE
            WHEN CURRENT_DATE >= next_due_date THEN true
            ELSE false
        END AS is_timeout,
    next_due_date - CURRENT_DATE AS days_until_due
   FROM collection.cheque_clearing_status ccs
  WHERE clearance_status = 'AWAITING_CLEARANCE'::collection.cheque_status;

COMMENT ON VIEW collection.pending_cheque_clearances IS 'Cheques awaiting clearance with timeout tracking';


-- collection.suspense_balances_by_policy source

CREATE OR REPLACE VIEW collection.suspense_balances_by_policy
AS SELECT policy_number,
    sum(
        CASE
            WHEN suspense_type = 'IR'::collection.suspense_type AND NOT is_reversed THEN amount
            ELSE 0::numeric
        END) AS ir_suspense_balance,
    sum(
        CASE
            WHEN suspense_type = 'CHEQUE_DISHONOR'::collection.suspense_type AND NOT is_reversed THEN amount
            ELSE 0::numeric
        END) AS dishonor_suspense_balance,
    sum(
        CASE
            WHEN suspense_type = 'DEFAULT_RELAPSE'::collection.suspense_type AND NOT is_reversed THEN amount
            ELSE 0::numeric
        END) AS default_suspense_balance,
    sum(
        CASE
            WHEN NOT is_reversed THEN amount
            ELSE 0::numeric
        END) AS total_suspense_balance,
    count(
        CASE
            WHEN NOT is_reversed THEN 1
            ELSE NULL::integer
        END) AS active_suspense_entries
   FROM collection.suspense_accounts sa
  GROUP BY policy_number;

COMMENT ON VIEW collection.suspense_balances_by_policy IS 'Suspense account balances by policy and type';
