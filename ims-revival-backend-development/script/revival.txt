-- DROP SCHEMA revival;

CREATE SCHEMA revival AUTHORIZATION postgres;


CREATE OR REPLACE FUNCTION revival.generate_ticket_id()
 RETURNS trigger
 LANGUAGE plpgsql
AS $function$
BEGIN
    IF NEW.ticket_id IS NULL THEN
        NEW.ticket_id := 'PSREYV' || LPAD(nextval('revival.ticket_id_seq')::TEXT, 9, '0');
    END IF;
    RETURN NEW;
END;
$function$
;

-- DROP FUNCTION revival.log_status_change();

CREATE OR REPLACE FUNCTION revival.log_status_change()
 RETURNS trigger
 LANGUAGE plpgsql
AS $function$
BEGIN
    IF (TG_OP = 'INSERT') OR (OLD.current_status IS DISTINCT FROM NEW.current_status) THEN
        INSERT INTO revival.status_change_history (
            request_id,
            from_status,
            to_status,
            changed_by
        ) VALUES (
            NEW.request_id,
            OLD.current_status,
            NEW.current_status,
            NEW.updated_at -- Using system time for simplicity
        );
    END IF;
    RETURN NEW;
END;
$function$
;

CREATE OR REPLACE VIEW revival.active_requests_summary
AS SELECT rr.request_id,
    rr.ticket_id,
    rr.policy_number,
    p.customer_name,
    rr.current_status,
    rr.indexed_date,
    rr.approval_date,
    rr.sla_end_date,
        CASE
            WHEN rr.sla_end_date IS NOT NULL THEN GREATEST(0::numeric, EXTRACT(day FROM rr.sla_end_date::timestamp with time zone - CURRENT_TIMESTAMP))
            ELSE NULL::numeric
        END AS days_until_sla_expiry,
    rr.number_of_installments,
    rr.installments_paid,
    rr.revival_amount,
    rr.installment_amount
   FROM revival.revival_requests rr
     JOIN common.policies p ON rr.policy_number::text = p.policy_number::text
  WHERE rr.current_status <> ALL (ARRAY['COMPLETED'::revival.request_status, 'REJECTED'::revival.request_status, 'TERMINATED'::revival.request_status, 'WITHDRAWN'::revival.request_status]);

COMMENT ON VIEW revival.active_requests_summary IS 'Active revival requests with SLA tracking';
